ASSET_REGISTER_STATUS = LET(
    assetId, TRIM(tblAssets[AssetID] & ""),
    assetName, TRIM(tblAssets[AssetName] & ""),
    assetType, TRIM(tblAssets[AssetType] & ""),
    status, TRIM(tblAssets[Status] & ""),
    linkedProject, TRIM(tblAssets[LinkedProjectID] & ""),
    entered, BYROW(
        HSTACK(
            assetId,
            assetName,
            assetType,
            TRIM(tblAssets[Site] & ""),
            TRIM(tblAssets[Location] & ""),
            TRIM(tblAssets[Department] & ""),
            TRIM(tblAssets[Owner] & ""),
            status,
            TRIM(tblAssets[InServiceDate] & ""),
            TRIM(tblAssets[Condition] & ""),
            TRIM(tblAssets[Criticality] & ""),
            TRIM(tblAssets[ReplacementCost] & ""),
            TRIM(tblAssets[UsefulLifeYears] & ""),
            TRIM(tblAssets[LastInspectionDate] & ""),
            TRIM(tblAssets[NextReviewDate] & ""),
            linkedProject,
            TRIM(tblAssets[Notes] & "")
        ),
        LAMBDA(r, SUM(--(r <> "")) > 0)
    ),
    enteredCount, SUM(--entered),
    namedCount, SUM(--((assetName <> "") * entered)),
    typedCount, SUM(--((assetType <> "") * entered)),
    statusCount, SUM(--((status <> "") * entered)),
    linkedCount, SUM(--((linkedProject <> "") * entered)),
    issueRows, IFERROR(DROP(Assets.ASSET_REGISTER_ISSUES, 1), ""),
    issueCount, IFERROR(SUM(--(TRIM(CHOOSECOLS(issueRows, 6) & "") <> "")), 0),
    VSTACK(
        {"Area", "Status", "Count", "Meaning"},
        HSTACK("Simple asset rows", IF(enteredCount > 0, "Ready", "Empty"), enteredCount, "Rows entered in tblAssets"),
        HSTACK("Named assets", IF(namedCount = enteredCount, "OK", "Needs review"), namedCount, "Entered rows with AssetName"),
        HSTACK("Typed assets", IF(typedCount = enteredCount, "OK", "Needs review"), typedCount, "Entered rows with AssetType"),
        HSTACK("Lifecycle statuses", IF(statusCount = enteredCount, "OK", "Needs review"), statusCount, "Entered rows with Status"),
        HSTACK("Linked project IDs", IF(linkedCount > 0, "Present", "Optional"), linkedCount, "Optional advisory links to current workbook planning rows"),
        HSTACK("Register issues", IF(issueCount > 0, "Needs review", "OK"), issueCount, "Missing IDs/names/statuses, duplicates, negative values, or advisory project links")
    )
);

ASSET_REGISTER_ISSUES = LET(
    assetId, TRIM(tblAssets[AssetID] & ""),
    assetName, TRIM(tblAssets[AssetName] & ""),
    assetType, TRIM(tblAssets[AssetType] & ""),
    status, TRIM(tblAssets[Status] & ""),
    linkedProject, TRIM(tblAssets[LinkedProjectID] & ""),
    replacementCostText, TRIM(tblAssets[ReplacementCost] & ""),
    usefulLifeText, TRIM(tblAssets[UsefulLifeYears] & ""),
    rows, get.GetBudgetDetailRows(),
    hdr, TAKE(rows, 1),
    data, DROP(rows, 1),
    projectKeys, VSTACK(
        TRIM(CHOOSECOLS(data, XMATCH("Source ID", hdr, 0)) & ""),
        TRIM(CHOOSECOLS(data, XMATCH("Job ID", hdr, 0)) & ""),
        TRIM(CHOOSECOLS(data, XMATCH("Project Description", hdr, 0)) & "")
    ),
    entered, BYROW(
        HSTACK(
            assetId,
            assetName,
            assetType,
            TRIM(tblAssets[Site] & ""),
            TRIM(tblAssets[Location] & ""),
            TRIM(tblAssets[Department] & ""),
            TRIM(tblAssets[Owner] & ""),
            status,
            TRIM(tblAssets[InServiceDate] & ""),
            TRIM(tblAssets[Condition] & ""),
            TRIM(tblAssets[Criticality] & ""),
            replacementCostText,
            usefulLifeText,
            TRIM(tblAssets[LastInspectionDate] & ""),
            TRIM(tblAssets[NextReviewDate] & ""),
            linkedProject,
            TRIM(tblAssets[Notes] & "")
        ),
        LAMBDA(r, SUM(--(r <> "")) > 0)
    ),
    duplicateId, IF(assetId = "", FALSE, COUNTIF(assetId, assetId) > 1),
    costValue, IFERROR(--replacementCostText, 0),
    usefulLifeValue, IFERROR(--usefulLifeText, 0),
    linkedProjectKnown, IF(linkedProject = "", TRUE, IFERROR(ISNUMBER(XMATCH(linkedProject, projectKeys, 0)), FALSE)),
    issue, IF(
        entered = FALSE,
        "",
        IF(
            assetId = "",
            "AssetID missing",
            IF(
                duplicateId,
                "Duplicate AssetID",
                IF(
                    assetName = "",
                    "AssetName missing",
                    IF(
                        status = "",
                        "Status missing",
                        IF(
                            costValue < 0,
                            "ReplacementCost negative",
                            IF(
                                usefulLifeValue < 0,
                                "UsefulLifeYears negative",
                                IF(
                                    linkedProjectKnown = FALSE,
                                    "LinkedProjectID not found in current workbook project keys (advisory)",
                                    ""
                                )
                            )
                        )
                    )
                )
            )
        )
    ),
    headers, {"SourceTable", "AssetID", "AssetName", "Status", "LinkedProjectID", "Issue"},
    queue, HSTACK(
        MAKEARRAY(ROWS(assetId), 1, LAMBDA(r, c, "tblAssets")),
        assetId,
        assetName,
        status,
        linkedProject,
        issue
    ),
    VSTACK(headers, IFERROR(FILTER(queue, entered * (issue <> "")), MAKEARRAY(1, COLUMNS(headers), LAMBDA(r, c, ""))))
);

PROJECT_PROMOTION_QUEUE = LET(
    rows, get.GetBudgetDetailRows(),
    hdr, TAKE(rows, 1),
    data, DROP(rows, 1),
    idxDesc, XMATCH("Project Description", hdr, 0),
    idxSource, XMATCH("Source ID", hdr, 0),
    idxJob, XMATCH("Job ID", hdr, 0),
    idxCategory, XMATCH("Category", hdr, 0),
    desc, TRIM(CHOOSECOLS(data, idxDesc) & ""),
    sourceId, TRIM(CHOOSECOLS(data, idxSource) & ""),
    jobId, TRIM(CHOOSECOLS(data, idxJob) & ""),
    category, TRIM(CHOOSECOLS(data, idxCategory) & ""),
    projectKey, IF(sourceId <> "", sourceId, IF(jobId <> "", jobId, desc)),
    candidateId, IF(jobId <> "", jobId, projectKey),
    constCol, LAMBDA(value, MAKEARRAY(ROWS(data), 1, LAMBDA(r, c, value))),
    mapped, IFERROR(COUNTIF(tblProjectAssetMap[ProjectKey], projectKey), 0) > 0,
    keep, (projectKey <> "") * NOT(mapped),
    headers, {
        "ProjectKey",
        "CandidateAssetId",
        "ProjectDescription",
        "AssetLabel",
        "AssetType",
        "ProposedChangeType",
        "SourceAssetId",
        "TargetAssetId",
        "InstalledState",
        "EvidenceId",
        "PromotionStatus",
        "ApplyReady",
        "ApplyStatus",
        "AppliedOn",
        "ApplyMessage"
    },
    queue, HSTACK(
        projectKey,
        candidateId,
        desc,
        IF(category <> "", category & " asset", desc),
        constCol("Asset"),
        constCol("new_asset"),
        constCol(""),
        candidateId,
        constCol("planned"),
        constCol(""),
        constCol("review"),
        constCol(FALSE),
        constCol(""),
        constCol(""),
        constCol("Review only; Office Script controls writes")
    ),
    VSTACK(headers, IFERROR(FILTER(queue, keep), MAKEARRAY(1, COLUMNS(headers), LAMBDA(r, c, ""))))
);

ASSET_MAPPING_ISSUES = LET(
    projectKey, TRIM(tblAssetMappingStaging[ProjectKey] & ""),
    changeType, LOWER(TRIM(tblAssetMappingStaging[ChangeType] & "")),
    sourceId, TRIM(tblAssetMappingStaging[SourceAssetId] & ""),
    targetId, TRIM(tblAssetMappingStaging[TargetAssetId] & ""),
    assetId, TRIM(tblAssetMappingStaging[AssetId] & ""),
    installedState, LOWER(TRIM(tblAssetMappingStaging[InstalledState] & "")),
    evidenceId, TRIM(tblAssetMappingStaging[EvidenceId] & ""),
    issue, MAP(
        projectKey,
        changeType,
        sourceId,
        targetId,
        assetId,
        installedState,
        evidenceId,
        LAMBDA(pk, ct, src, tgt, asset, inst, evid,
            TEXTJOIN(
                "; ",
                TRUE,
                IF(pk = "", "ProjectKey missing", ""),
                IF(ct = "", "ChangeType missing", ""),
                IF(AND(ct = "new_asset", tgt = ""), "new_asset requires TargetAssetId", ""),
                IF(AND(ct = "replace_asset", OR(src = "", tgt = "")), "replace_asset requires SourceAssetId and TargetAssetId", ""),
                IF(AND(ct = "upgrade_asset", src <> tgt), "upgrade_asset requires SourceAssetId = TargetAssetId", ""),
                IF(AND(inst = "installed", evid = ""), "installed asset requires EvidenceId", ""),
                IF(AND(asset = "", tgt = "", src = ""), "Asset identity missing", "")
            )
        )
    ),
    headers, {"SourceTable", "ProjectKey", "ChangeType", "SourceAssetId", "TargetAssetId", "Issue"},
    constCol, LAMBDA(value, MAKEARRAY(ROWS(projectKey), 1, LAMBDA(r, c, value))),
    queue, HSTACK(constCol("tblAssetMappingStaging"), projectKey, changeType, sourceId, targetId, issue),
    VSTACK(headers, IFERROR(FILTER(queue, issue <> ""), MAKEARRAY(1, COLUMNS(headers), LAMBDA(r, c, ""))))
);

REPLACEMENT_SOURCE_TARGET_ISSUES = LET(
    mappingSource, MAKEARRAY(ROWS(tblAssetMappingStaging[ProjectKey]), 1, LAMBDA(r, c, "tblAssetMappingStaging")),
    changeSource, MAKEARRAY(ROWS(tblAssetChanges[ProjectKey]), 1, LAMBDA(r, c, "tblAssetChanges")),
    mappingRows, HSTACK(
        mappingSource,
        TRIM(tblAssetMappingStaging[ProjectKey] & ""),
        LOWER(TRIM(tblAssetMappingStaging[ChangeType] & "")),
        TRIM(tblAssetMappingStaging[SourceAssetId] & ""),
        TRIM(tblAssetMappingStaging[TargetAssetId] & "")
    ),
    changeRows, HSTACK(
        changeSource,
        TRIM(tblAssetChanges[ProjectKey] & ""),
        LOWER(TRIM(tblAssetChanges[ChangeType] & "")),
        TRIM(tblAssetChanges[SourceAssetId] & ""),
        TRIM(tblAssetChanges[TargetAssetId] & "")
    ),
    rows, VSTACK(mappingRows, changeRows),
    changeType, CHOOSECOLS(rows, 3),
    sourceId, CHOOSECOLS(rows, 4),
    targetId, CHOOSECOLS(rows, 5),
    issue, MAP(
        changeType,
        sourceId,
        targetId,
        LAMBDA(ct, src, tgt,
            TEXTJOIN(
                "; ",
                TRUE,
                IF(AND(ct = "replace_asset", OR(src = "", tgt = "")), "replace_asset requires source and target", ""),
                IF(AND(ct = "upgrade_asset", src <> tgt), "upgrade_asset source and target must match", "")
            )
        )
    ),
    headers, {"SourceTable", "ProjectKey", "ChangeType", "SourceAssetId", "TargetAssetId", "Issue"},
    queue, HSTACK(rows, issue),
    VSTACK(
        headers,
        IFERROR(
            FILTER(queue, (issue <> "") * (((changeType = "replace_asset") + (changeType = "upgrade_asset")) > 0)),
            MAKEARRAY(1, COLUMNS(headers), LAMBDA(r, c, ""))
        )
    )
);

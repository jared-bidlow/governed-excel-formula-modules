        Rows, DROP(Axis, 1),
        GroupLabel, INDEX(Header, 1, 3),
        ScreenCols, COLUMNS(Header),
        PadTo, LAMBDA(arr,
            IF(
                COLUMNS(arr) < ScreenCols,
                HSTACK(
                    arr,
                    MAKEARRAY(ROWS(arr), ScreenCols - COLUMNS(arr), LAMBDA(rw, c, ""))
                ),
                arr
            )

        ActionTotalsRaw, GROUPBY(
            CHOOSECOLS(Rows, 2),
            DecisionDollars,
            SUM,
            0,
            0,
            1
        ),
        ActionTotals, SORTBY(ActionTotalsRaw, CHOOSECOLS(ActionTotalsRaw, 2), -1),
        GrandTotalRow, HSTACK("Grand Total", SUM(DecisionDollars)),
        GroupJobActionPivot, PIVOTBY(

                VSTACK(GroupLabel, GroupVals),

        KeyHdr, INDEX(AxisHdr, 1, 1),

        ScopeStateRaw, CHOOSECOLS(AxisRows, 9),
        BucketVals, IF(
            ScopeStateRaw = "In Scope",
            "1 In Scope",
            IF(
                ScopeStateRaw = "Hidden by Future",
                "2 Hidden by Future",
                "3 Hidden by Closed"
            )
        ),
        Rem, CHOOSECOLS(AxisRows, 6),
        Burn, CHOOSECOLS(AxisRows, 5),
        Proj, CHOOSECOLS(AxisRows, 4),
        Jobs, CHOOSECOLS(AxisRows, 11),
        KeepMask, 1 + 0 * Jobs,
        BaseRows, DROP(Base, 1),
        BaseKeys, CHOOSECOLS(BaseRows, 1),
        Num0, LAMBDA(arr, IFERROR(0 + arr, 0)),
        SafePct, LAMBDA(num, den,
            LET(
                NumVal, Num0(num),
                DenVal, Num0(den),
                IF(
                    DenVal = 0,
                    IF(NumVal <> 0, 1, 0),
                    NumVal / DenVal
                )
            )
        ),
        AllTotalsRaw, GROUPBY(
            KeyVals,
            HSTACK(Proj, Rem),
            HSTACK(SUM, SUM),
            0,
            0,
            1
        ),

            BaseKeys,
            CHOOSECOLS(AllTotalsRaw, 1),
            CHOOSECOLS(AllTotalsRaw, 3),

            3,
            0,
            1,
            0,
            1,
            VSTACK(1, KeepMask),
            1

        FnHdrs, DROP(CHOOSEROWS(PivotRaw, 3), , 1),
        ValHdrs, DROP(CHOOSEROWS(PivotRaw, 4), , 1),
        PivotRows, DROP(PivotRaw, 4),
        PivotKeys, CHOOSECOLS(PivotRows, 1),
        PivotVals, DROP(PivotRows, , 1),
        CleanBucket, LAMBDA(x, IFERROR(TEXTAFTER(x, " "), x)),
        MetricIdx, LAMBDA(fn, val,
            FILTER(SEQUENCE(, COLUMNS(PivotVals)), (FnHdrs = fn) * (ValHdrs = val))
        ),
        MetricVals, LAMBDA(fn, val,
            Num0(XLOOKUP(BaseKeys, PivotKeys, CHOOSECOLS(PivotVals, MetricIdx(fn, val)), 0))
        ),
        MetricHdr, LAMBDA(fn, val, label,
            MAP(CHOOSECOLS(BucketHdrs, MetricIdx(fn, val)), LAMBDA(x, CleanBucket(x) & " | " & label))
        ),
        ProjVals, MetricVals("SUM", "Projected"),
        RemVals, MetricVals("SUM", "Remaining"),
        JobVals, MetricVals("SUM", "Jobs"),
        BurnVals, MetricVals("SUM", "Burned YTD"),

        HiddenPctVals, MetricVals("PERCENTOF", "Hidden Dollars Share"),
        TotalProjBuckets, BYCOL(ProjVals, LAMBDA(col, SUM(col))),
        TotalRemBuckets, BYCOL(RemVals, LAMBDA(col, SUM(col))),
        TotalJobBuckets, BYCOL(JobVals, LAMBDA(col, SUM(col))),
        TotalBurnBuckets, BYCOL(BurnVals, LAMBDA(col, SUM(col))),
        TotalHiddenBuckets, BYCOL(HiddenVals, LAMBDA(col, SUM(col))),
        AvgHdr, MetricHdr("SUM", "Burned YTD", "Avg Burn / Job"),
        RemShareHdr, MetricHdr("PERCENTOF", "Remaining Share", "Remaining Share %"),
        HiddenHdr, MetricHdr("PERCENTOF", "Hidden Dollars Share", "Hidden Dollars %"),
        HdrOut, HSTACK(
            KeyHdr,
            MetricHdr("SUM", "Remaining", "Remaining"),
            MetricHdr("SUM", "Jobs", "Jobs"),
            AvgHdr,
            RemShareHdr,
            HiddenHdr
        ),
        RowsOut, HSTACK(
            BaseKeys,
            RemVals,
            JobVals,
            SafePct(BurnVals, JobVals),
            RemShareVals,
            HiddenPctVals
        ),
        TotalRow, HSTACK(
            "Grand Total",
            TotalRemBuckets,
            TotalJobBuckets,
            SafePct(TotalBurnBuckets, TotalJobBuckets),
            SafePct(TotalRemBuckets, SUM(BaseRemAll)),
            SafePct(TotalHiddenBuckets, SUM(TotalHiddenBuckets))
        ),
        IF(
            ROWS(BaseKeys) = 0,
            HdrOut,
            VSTACK(HdrOut, RowsOut, TotalRow)
        )
    )
);

FINANCE_READINESS_STATUS = LET(
    classifiedRows, IFERROR(DROP(AssetFinance.CLASSIFIED_MODEL_INPUTS, 1), ""),
    evidenceId, IFERROR(CHOOSECOLS(classifiedRows, 1), ""),
    amount, IFERROR(CHOOSECOLS(classifiedRows, 7), 0),
    classifiedCount, IFERROR(SUM(--(TRIM(evidenceId & "") <> "")), 0),
    totalClassifiedAmount, IFERROR(SUM(amount), 0),
    depRows, IFERROR(DROP(AssetFinance.DEPRECIATION_SCHEDULE, 1), ""),
    fundRows, IFERROR(DROP(AssetFinance.FUNDING_REQUIREMENTS, 1), ""),
    depIssueCount, IFERROR(SUM(--(TRIM(CHOOSECOLS(depRows, 12) & "") <> "")), 0),
    fundingIssueCount, IFERROR(SUM(--(TRIM(CHOOSECOLS(fundRows, 7) & "") <> "")), 0),
    annualDepreciation, IFERROR(SUM(CHOOSECOLS(depRows, 9)), 0),
    fundingAmount, IFERROR(SUM(CHOOSECOLS(fundRows, 6)), 0),
    VSTACK(
        {"Metric", "Value", "Meaning"},
        HSTACK("Finance-ready evidence count", classifiedCount, "Rows with PresentWithClassifiedEvidence = TRUE"),
        HSTACK("Unsupported depreciation methods", depIssueCount, "Rows with DepreciationIssue text"),
        HSTACK("Unsupported funding rules", fundingIssueCount, "Rows with FundingIssue text"),
        HSTACK("Total classified amount", totalClassifiedAmount, "Amount from classified model inputs"),
        HSTACK("Annual depreciation", annualDepreciation, "Supported straight-line annual depreciation"),
        HSTACK("Funding requirement amount", fundingAmount, "Supported full grouped classified funding amount")
    )
);

CLASSIFIED_MODEL_INPUTS = LET(
    Headers, {
        "EvidenceId",
        "ProjectKey",
        "AssetId",
        "AssetLabel",
        "AssetType",
        "EvidenceDate",
        "Amount",
        "FundingSource",
        "DepreciationClass",
        "ClassifiedCategoryId",
        "ClassifiedCategoryName",
        "PresentWithClassifiedEvidence",
        "ReviewIssue"
    },
    Flag, tblAssetEvidence_ModelInputs[PresentWithClassifiedEvidence],
    Keep, (Flag = TRUE) + (UPPER(TRIM(Flag & "")) = "TRUE") > 0,
    SourceRows, HSTACK(
        tblAssetEvidence_ModelInputs[EvidenceId],
        tblAssetEvidence_ModelInputs[ProjectKey],
        tblAssetEvidence_ModelInputs[AssetId],
        tblAssetEvidence_ModelInputs[AssetLabel],
        tblAssetEvidence_ModelInputs[AssetType],
        tblAssetEvidence_ModelInputs[EvidenceDate],
        IFERROR(--tblAssetEvidence_ModelInputs[Amount], 0),
        tblAssetEvidence_ModelInputs[FundingSource],
        tblAssetEvidence_ModelInputs[DepreciationClass],
        tblAssetEvidence_ModelInputs[ClassifiedCategoryId],
        tblAssetEvidence_ModelInputs[ClassifiedCategoryName],
        Flag,
        tblAssetEvidence_ModelInputs[ReviewIssue]
    ),
    VSTACK(
        Headers,
        IFERROR(
            FILTER(SourceRows, Keep),
            MAKEARRAY(1, COLUMNS(Headers), LAMBDA(r, c, ""))
        )
    )
);

DEPRECIATION_SCHEDULE = LET(
    Headers, {
        "EvidenceId",
        "ProjectKey",
        "AssetId",
        "AssetLabel",
        "DepreciationClass",
        "DepreciationMethod",
        "UsefulLifeYears",
        "DepreciableAmount",
        "AnnualDepreciation",
        "EvidenceDate",
        "ClassifiedCategoryName",
        "DepreciationIssue"
    },
    IFERROR(
        LET(
            Rows, FILTER(
                DROP(AssetFinance.CLASSIFIED_MODEL_INPUTS, 1),
                CHOOSECOLS(DROP(AssetFinance.CLASSIFIED_MODEL_INPUTS, 1), 1) <> ""
            ),
            EvidenceId, CHOOSECOLS(Rows, 1),
            ProjectKey, CHOOSECOLS(Rows, 2),
            AssetId, CHOOSECOLS(Rows, 3),
            AssetLabel, CHOOSECOLS(Rows, 4),
            EvidenceDate, CHOOSECOLS(Rows, 6),
            Amount, CHOOSECOLS(Rows, 7),
            DepreciationClass, CHOOSECOLS(Rows, 9),
            CategoryName, CHOOSECOLS(Rows, 11),
            RawMethod, IFERROR(
                XLOOKUP(
                    DepreciationClass,
                    tblAssetFinanceAssumptions[DepreciationClass],
                    tblAssetFinanceAssumptions[DepreciationMethod],
                    ""
                ),
                ""
            ),
            Method, IF(TRIM(RawMethod & "") = "", "Straight-line", RawMethod),
            SupportedMethod, LOWER(TRIM(Method & "")) = "straight-line",
            UsefulLifeRaw, IFERROR(
                --XLOOKUP(
                    DepreciationClass,
                    tblAssetFinanceAssumptions[DepreciationClass],
                    tblAssetFinanceAssumptions[UsefulLifeYears],
                    10
                ),
                10
            ),
            UsefulLife, IF(UsefulLifeRaw > 0, UsefulLifeRaw, 10),
            AnnualDepreciation, IF(SupportedMethod, Amount / UsefulLife, ""),
            DepreciationIssue, IF(SupportedMethod, "", "Unsupported DepreciationMethod: " & Method),
            VSTACK(
                Headers,
                HSTACK(
                    EvidenceId,
                    ProjectKey,
                    AssetId,
                    AssetLabel,
                    DepreciationClass,
                    Method,
                    UsefulLife,
                    Amount,
                    AnnualDepreciation,
                    EvidenceDate,
                    CategoryName,
                    DepreciationIssue
                )
            )
        ),
        VSTACK(Headers, MAKEARRAY(1, COLUMNS(Headers), LAMBDA(r, c, "")))
    )
);

FUNDING_REQUIREMENTS = LET(
    Headers, {
        "FundingSource",
        "ProjectKey",
        "AssetId",
        "ClassifiedCategoryName",
        "FundingRequirementRule",
        "FundingRequirementAmount",
        "FundingIssue"
    },
    IFERROR(
        LET(
            Rows, FILTER(
                DROP(AssetFinance.CLASSIFIED_MODEL_INPUTS, 1),
                CHOOSECOLS(DROP(AssetFinance.CLASSIFIED_MODEL_INPUTS, 1), 1) <> ""
            ),
            ProjectKey, CHOOSECOLS(Rows, 2),
            AssetId, CHOOSECOLS(Rows, 3),
            Amount, CHOOSECOLS(Rows, 7),
            FundingSource, IF(
                TRIM(CHOOSECOLS(Rows, 8) & "") = "",
                "(No Funding Source)",
                CHOOSECOLS(Rows, 8)
            ),
            CategoryName, IF(
                TRIM(CHOOSECOLS(Rows, 11) & "") = "",
                "(Unclassified)",
                CHOOSECOLS(Rows, 11)
            ),
            Keys, UNIQUE(HSTACK(FundingSource, ProjectKey, AssetId, CategoryName)),
            RawRules, IFERROR(
                XLOOKUP(
                    CHOOSECOLS(Keys, 1),
                    tblAssetFinanceAssumptions[FundingSource],
                    tblAssetFinanceAssumptions[FundingRequirementRule],
                    ""
                ),
                ""
            ),
            Rules, IF(TRIM(RawRules & "") = "", "Fund full classified amount", RawRules),
            SupportedRule, LOWER(TRIM(Rules & "")) = "fund full classified amount",
            GroupedAmount, MAP(
                CHOOSECOLS(Keys, 1),
                CHOOSECOLS(Keys, 2),
                CHOOSECOLS(Keys, 3),
                CHOOSECOLS(Keys, 4),
                LAMBDA(fs, pk, aid, cat,
                    SUM(FILTER(Amount, (FundingSource = fs) * (ProjectKey = pk) * (AssetId = aid) * (CategoryName = cat), 0))
                )
            ),
            RequirementAmount, IF(SupportedRule, GroupedAmount, ""),
            FundingIssue, IF(SupportedRule, "", "Unsupported FundingRequirementRule: " & Rules),
            VSTACK(Headers, HSTACK(Keys, Rules, RequirementAmount, FundingIssue))
        ),
        VSTACK(Headers, MAKEARRAY(1, COLUMNS(Headers), LAMBDA(r, c, "")))
    )
);

FINANCE_TOTALS = LET(
    Headers, {"Metric", "Group", "Amount"},
    IFERROR(
        LET(
            Rows, DROP(AssetFinance.CLASSIFIED_MODEL_INPUTS, 1),
            EvidenceId, CHOOSECOLS(Rows, 1),
            AssetId, CHOOSECOLS(Rows, 3),
            Amount, CHOOSECOLS(Rows, 7),
            DepRows, DROP(AssetFinance.DEPRECIATION_SCHEDULE, 1),
            FundRows, DROP(AssetFinance.FUNDING_REQUIREMENTS, 1),
            VSTACK(
                Headers,
                HSTACK("Classified evidence amount", "All classified rows", SUM(Amount)),
                HSTACK("Annual depreciation", "Straight-line default", SUM(CHOOSECOLS(DepRows, 9))),
                HSTACK("Funding requirement amount", "Grouped requirements", SUM(CHOOSECOLS(FundRows, 6))),
                HSTACK("Classified evidence count", "All classified rows", SUM(--(EvidenceId <> ""))),
                HSTACK("Classified asset count", "Unique assets", IFERROR(ROWS(UNIQUE(FILTER(AssetId, AssetId <> ""))), 0))
            )
        ),
        VSTACK(Headers, MAKEARRAY(1, COLUMNS(Headers), LAMBDA(r, c, "")))
    )
);

CHART_FEEDS = LET(
    Headers, {"FeedName", "Series", "Category", "Value"},
    IFERROR(
        LET(
            FundingRows, IFERROR(
                LET(
                    FundSourceRows, DROP(AssetFinance.FUNDING_REQUIREMENTS, 1),
                    FundRows, FILTER(
                        FundSourceRows,
                        TRIM(CHOOSECOLS(FundSourceRows, 6) & "") <> ""
                    ),
                    FundingAmount, CHOOSECOLS(FundRows, 6),
                    FundingSource, CHOOSECOLS(FundRows, 1),
                    ChartGroup, IFERROR(
                        XLOOKUP(
                            FundingSource,
                            tblAssetFinanceAssumptions[FundingSource],
                            tblAssetFinanceAssumptions[ChartGroup],
                            FundingSource
                        ),
                        FundingSource
                    ),
                    FundingKeys, UNIQUE(ChartGroup),
                    FundingAmounts, MAP(
                        FundingKeys,
                        LAMBDA(groupName, SUM(FILTER(FundingAmount, ChartGroup = groupName, 0)))
                    ),
                    HSTACK(
                        MAKEARRAY(ROWS(FundingKeys), 1, LAMBDA(r, c, "Funding by chart group")),
                        MAKEARRAY(ROWS(FundingKeys), 1, LAMBDA(r, c, "Funding requirement")),
                        FundingKeys,
                        FundingAmounts
                    )
                ),
                MAKEARRAY(1, COLUMNS(Headers), LAMBDA(r, c, ""))
            ),
            DepRowsOut, IFERROR(
                LET(
                    DepSourceRows, DROP(AssetFinance.DEPRECIATION_SCHEDULE, 1),
                    DepRows, FILTER(
                        DepSourceRows,
                        TRIM(CHOOSECOLS(DepSourceRows, 9) & "") <> ""
                    ),
                    DepClass, IF(
                        TRIM(CHOOSECOLS(DepRows, 5) & "") = "",
                        "(No Depreciation Class)",
                        CHOOSECOLS(DepRows, 5)
                    ),
                    AnnualDepreciation, CHOOSECOLS(DepRows, 9),
                    DepKeys, UNIQUE(DepClass),
                    DepAmounts, MAP(
                        DepKeys,
                        LAMBDA(className, SUM(FILTER(AnnualDepreciation, DepClass = className, 0)))
                    ),
                    HSTACK(
                        MAKEARRAY(ROWS(DepKeys), 1, LAMBDA(r, c, "Annual depreciation by class")),
                        MAKEARRAY(ROWS(DepKeys), 1, LAMBDA(r, c, "Annual depreciation")),
                        DepKeys,
                        DepAmounts
                    )
                ),
                MAKEARRAY(1, COLUMNS(Headers), LAMBDA(r, c, ""))
            ),
            CombinedRows, VSTACK(FundingRows, DepRowsOut),
            VSTACK(
                Headers,
                IFERROR(
                    FILTER(CombinedRows, TRIM(CHOOSECOLS(CombinedRows, 1) & "") <> ""),
                    MAKEARRAY(1, COLUMNS(Headers), LAMBDA(r, c, ""))
                )
            )
        ),
        VSTACK(Headers, MAKEARRAY(1, COLUMNS(Headers), LAMBDA(r, c, "")))
    )
);
